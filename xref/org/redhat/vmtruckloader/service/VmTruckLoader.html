57  	private static final int INVALID_COMMAND_INPUT = 1;

59  	private static final int PROGRAM_THROWN_EXCEPTION = 3;

64  	public static void runWithCatch(String[] args) {
65  		try {
66  			run(args);
67  		} catch ( Throwable t ) {
68  			consolePrint(t.getMessage());
69  			if ( t.getCause() != null )
70  				consolePrint(t.getCause().getMessage());
71  			System.exit(PROGRAM_THROWN_EXCEPTION);
72  		}
73  	}

75  	public static void run(String[] args) {
76  		Arguments arguments = setDefaultValues(validateArgs(VmTruckLoader.extractParameters(args)));
77  
78  		List<MachineSpecification> specs;
79  		if ( arguments.getFile() != null )
80  			specs = turnFileIntoSpec(arguments.getFile());
81  		else {
82  			specs = new ArrayList<MachineSpecification>();
83  			specs.add(turnLineIntoSpec(arguments.getLine()));
84  		}
85  		
86  		Properties props = System.getProperties();
87  		props.put("org.slf4j.simpleLogger.defaultLogLevel", "ERROR");
88  		System.setProperties(props);
89  
90  		WeldContainer weld = new Weld().initialize();
91  		MachineService machineService = weld.instance().select(MachineService.class).get();
92  		machineService.setConnectionData(arguments.getUsername(), arguments.getPassword(), URLUtils.createUrl(arguments.getVCenterURl()));
93  
94  		if ( specs.isEmpty() ) {
95  			consolePrint("No machine specification found.");
96  			System.exit(0);
97  		}
98  		
99  		validatePostExecScript(arguments.getPostExec());
100 		
101 		for ( MachineSpecification spec : specs ) {
102 			switch (arguments.getAction()) {
103 			case CREATE:
104 				StringUtil.checkIfStringValid(arguments.getTemplateName(),"The template name is required for action " + arguments.getAction());
105 				consolePrint("Creating machine " + spec.getVmName() + " ...");
106 				spec = machineService.cloneMachine(arguments.getTemplateName(), spec);
107 				spec.setMAC("@MAC");
108 				consolePrint("- machine " + spec.getVmName() + " has been successfully created." );
109 				consolePrint("-      with @MAC: " +	spec.getMAC());	
110 				updateLineInCsvFile(arguments.getFile().getAbsolutePath(), spec);
111 				runPostExecScriptIfAny(arguments.getPostExec(), spec);
112 				break;
113 			case DELETE:
114 				consolePrint("Deleting VM" + spec.getHostname());
115 				machineService.deleteMachine(spec.getHostname());
116 				break;
117 			case START:
118 				consolePrint("Starting VM" + spec.getHostname());
119 				machineService.startMachine(spec.getHostname());
120 				break;
121 			case STOP:
122 				consolePrint("Stopping VM" + spec.getHostname());
123 				machineService.stopMachine(spec.getHostname());
124 				break;
125 			case RESTART:
126 				consolePrint("Restarting VM" + spec.getHostname());
127 				machineService.stopMachine(spec.getHostname());
128 				machineService.startMachine(spec.getHostname());
129 				break;
130 			case EDIT:				
131 				consolePrint("Edit VM" + spec.getHostname());
132 				machineService.editMachine(spec);
133 				break;
134 			default:
135 				throw new IllegalStateException("Unrecognized action requested:" + arguments.getAction());
136 			}
137 		}
138 	}

140 	private static void runPostExecScriptIfAny(File postExec,
141 			MachineSpecification spec) {
142 		if ( postExec == null) return; 
143 		for ( String line : executeScriptAndReturnsResult(buildCommandLine(postExec,spec)))
144 			consolePrint(line);
145 	}
146 
147 	private static String buildCommandLine(File postExec,
148 			MachineSpecification spec) {
149 		
150 		return postExec.getAbsolutePath() + COMMAND_LINE_SEPARATOR + spec.getHostname() + COMMAND_LINE_SEPARATOR + 
151 				spec.getMAC() + COMMAND_LINE_SEPARATOR + spec.getRole() + COMMAND_LINE_SEPARATOR + spec.getRole();
152 	}
153 
154 	private static void validatePostExecScript(File postExec) {
155 		if ( postExec == null) return; 
156 
157 		if ( ! postExec.exists() )
158 	      throw new IllegalArgumentException(postExec.getAbsolutePath() + " does not exist, and therefore can't be run ");
159 		
160 		if ( ! postExec.canExecute() )
161 		      throw new IllegalArgumentException(postExec.getAbsolutePath() + " is not executable, and therefore can't be run ");
162 	}

164 	private static List<MachineSpecification> turnFileIntoSpec(File file) {
165 		List<String> lines =  FileUtils.readLineByLine(file);
166 		List<MachineSpecification> specs = new ArrayList<MachineSpecification>(lines.size());
167 		for ( String line : lines ) {
168 			specs.add(turnLineIntoSpec(line));
169 		}
170 		return specs;
171 	}

189 		if ( printCarriageReturn )

199 	private static Arguments extractParameters(String[] args) {
200 		Arguments arguments = new Arguments();
201 		JCommander jcommander = null;
202 		try {
203 			jcommander = new JCommander(arguments, args);
204 			jcommander.setProgramName(PROG_NAME);
205 			if (arguments.isHelp()) {
206 				jcommander.usage();
207 				System.exit(0);
208 			}
209 
210 			if ( arguments.getLine() == null && arguments.getFile() == null ) {
211 				consolePrint(PROG_NAME + " requires either a line or a file:");
212 				jcommander.usage();
213 				System.exit(INPUT_ERROR_NOFILE_NORLINE);
214 			}
215 
216 		} catch (ParameterException e) {
217 			System.out.println(e.getMessage());
218 			System.exit(INVALID_COMMAND_INPUT);
219 		}
220 		return arguments;
221 	}
